        TelaEstoque tela = new TelaEstoque();
        String Quantidade= jTextFieldQNT.getText();
        String codigo= jLabelCodigo.getText();
        int cod_produto = Integer.parseInt(codigo);
        int qnt_produto = Integer.parseInt(Quantidade);
        EstoqueDAO estoque = new EstoqueDAO();
